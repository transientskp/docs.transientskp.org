import math

def nice_format(f):
    if f > 9999 or f < 0.01:
        return "%.2e" % f
    else:
        return "%.2f" % f


def substitute(value, sub, test_f):
    try:
        if test_f(value):
            return sub
    except TypeError:
        pass
    return value
